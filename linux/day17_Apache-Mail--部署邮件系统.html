部署邮件系统

一、DNS MX
tianyun.com. IN MX 10 mail.tianyun.com.
mail.tianyun.com. IN A 192.168.10.66
[root@yangs ~]＃ host -t MX tianyun.com

二、MTA：Postfix
1. Postfix软件包安装
[root@yangs ~]# rpm -q postfix
postfix-2.6.6-2.2.el6_1.x86_64
[root@yangs ~]# netstat -tnlp |grep :25
tcp 0 0 127.0.0.1:25 0.0.0.0:* LISTEN 1889/master
tcp 0 0 ::1:25 :::* LISTEN 1889/master

2. 基本配置
[root@yangs ~]# vim /etc/postfix/main.cf
myhostname = mail.tianyun.com
mydomain = tianyun.com
myorigin = $mydomain
inet_interfaces = all
mydestination = $myhostname, localhost, $mydomain

[root@yangs ~]# service postfix restart
[root@yangs ~]# netstat -tnlp |grep :25
tcp 0 0 0.0.0.0:25 0.0.0.0:* LISTEN 28266/master
tcp 0 0 :::25 :::* LISTEN 28266/master

alias_maps = hash:/etc/aliases

hr: hr1,hr2,hr3

6. 实现SMTP身份认证
由于Postfix默认只为信任的网段转发 (mynetworks = 127.0.0.1)
实现SMTP身份认证功能后，可以为通过身份认证的用户转发邮件

默认不支持SMTP认证
[root@yangs ~]# telnet 192.168.10.66 25
Trying 192.168.10.66...
Connected to 192.168.10.66.
Escape character is '^]'.
220 mail.tianyun.com ESMTP Postfix
ehlo hi
250-mail.tianyun.com
250-PIPELINING
250-SIZE 10240000
250-VRFY
250-ETRN
250-ENHANCEDSTATUSCODES
250-8BITMIME
250 DSN

a. saslauthd
[root@yangs ~]# yum -y install cyrus-sasl*
[root@yangs ~]# /etc/init.d/saslauthd start
[root@yangs ~]# chkconfig saslauthd on

b. posfix： /etc/postfix/main.cf
smtpd_sasl_auth_enable = yes
smtpd_sasl_security_options = noanonymous
smtpd_recipient_restrictions = permit_mynetworks,
permit_sasl_authenticated,
reject_unauth_destination

[root@yangs ~]# service postfix restart

[root@yangs ~]# telnet 192.168.10.137 25
Trying 192.168.10.137...
Connected to 192.168.10.137.
Escape character is '^]'.

250-PIPELINING
250-SIZE 10240000

三、配置MRA，提供POP或IMAP
[root@yangs ~]# yum -y install dovecot
[root@yangs ~]# vim /etc/dovecot/dovecot.conf
protocols = pop3

允许明文认证 vs SSL实现加密连接
[root@yangs ~]# vim /etc/dovecot/conf.d/10-auth.conf
disable_plaintext_auth = no

指定用户邮箱的类型和位置
[root@yangs ~]# vim /etc/dovecot/conf.d/10-mail.conf
mail_location = mbox:~/mail:INBOX=/var/mail/%u、
