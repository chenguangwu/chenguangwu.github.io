增加内核模块



二、给现有的内核增加功能（模块）
注：使用和当前内核版本完全匹配内核源码包
1. 查看当前内核版本
[root@tianyun ~]# uname -r
2.6.32-358.el6.x86_64

2. 下载内核源码
ftp://ftp.redhat.com/pub/redhat/linux/enterprise/6Server
kernel-2.6.32-358.el6.src.rpm

3. 需要工具rpm-build，开发库，ncurses-devel
[root@tianyun ~]# rpm -q rpm-build
rpm-build-4.8.0-32.el6.x86_64

4. 产生kernel源码
//释放src.rpm
[root@tianyun ~]# rpm -ivh kernel-2.6.32-358.el6.src.rpm
[root@tianyun ~]# ls rpmbuild/
SOURCES SPECS
[root@tianyun ~]# ls rpmbuild/SOURCES/
//将源码解压
[root@tianyun ~]# rpmbuild -bp --target=$(uname -m) ~/rpmbuild/SPECS/kernel.spec
[root@tianyun ~]# ls rpmbuild/
BUILD BUILDROOT RPMS SOURCES SPECS SRPMS
[root@tianyun ~]# cd ~/rpmbuild/BUILD/kernel-2.6.32-358.el6/linux-2.6.32-358.el6.x86_64/


例1：让kernel支持jfs文件系统（模块 jfs.ko）
以下的操作在内核的源码目录
1. 编译模块
[root@LiangPad linux-2.6.32-358.el6.x86_64]# make menuconfig
<M> Reiserfs support │ │
│ │ [ ] Enable reiserfs debug mode (NEW) │ │
│ │ [ ] Stats in /proc/fs/reiserfs (NEW) │ │
│ │ [ ] ReiserFS extended attributes (NEW) │ │
<M> JFS filesystem support │ │
│ │ [ ] JFS POSIX Access Control Lists (NEW) │ │
│ │ [ ] JFS Security Labels (NEW) │ │
│ │ [ ] JFS debugging (NEW) │ │
│ │ [ ] JFS statistics (NEW) │ │
[root@LiangPad linux-2.6.32-358.el6.x86_64]# make fs/jfs/jfs.ko //编译模块
[root@LiangPad linux-2.6.32-358.el6.x86_64]# make fs/reiserfs/reiserfs.ko //编译模块

2. 复制模块到当前内核模块目录/lib/modules
[root@LiangPad linux-2.6.32-358.el6.x86_64]# mkdir /lib/modules/$(uname -r)/kernel/fs/{reiserfs,jfs}
[root@LiangPad linux-2.6.32-358.el6.x86_64]# cp fs/jfs/jfs.ko /lib/modules/2.6.32-358.el6.x86_64/kernel/fs/jfs/
[root@LiangPad linux-2.6.32-358.el6.x86_64]# cp fs/reiserfs/reiserfs.ko /lib/modules/2.6.32-358.el6.x86_64/kernel/fs/reiserfs/

3. 重新生成模块的依赖关系
[root@LiangPad ~]# ls /lib/modules/2.6.32-358.el6.x86_64/
build modules.ccwmap modules.modesetting modules.symbols.bin
extra modules.dep modules.networking modules.usbmap
kernel modules.dep.bin modules.ofmap source
[root@tianyun 2.6.18-308.el5]# depmod

扩展知识：
创建jfs、reiserfs文件系统
1. 安装创建jfs文件系统的软件
[root@tianyun ~]# tar xvf jfsutils-1.1.15.tar.gz
[root@tianyun ~]# cd jfsutils-1.1.15
[root@tianyun jfsutils-1.1.15]# ./configure && make && make install
2. 准备分区或卷
3. 创建文件系统
[root@tianyun ~]# mkfs.jfs /dev/sda5
[root@tianyun ~]# mkreiserfs /dev/sda6
[root@tianyun ~]# mkdir /mnt/jfs
[root@tianyun ~]# mount /dev/sda5 /mnt/jfs
[root@tianyun ~]# lsmod |grep jfs
jfs 149228 1

管理模块
modinfo 查看模块的信息 modinfo jfs modinfo -n jfs
lsmod 查看已加载的模块
modprobe jfs 手动加载模块
modprobe -r xfs 移出模块
