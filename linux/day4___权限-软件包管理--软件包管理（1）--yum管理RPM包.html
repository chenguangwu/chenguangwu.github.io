第一种情况：使用系统光盘作为YUM源（本地源）安装rpm软件
1. 指定YUM源
/etc/yum.repo.d/*.repo
[root@tianyun ~]# cd /etc/yum.repos.d/
[root@tianyun yum.repos.d]# vim rhel6.repo //文件名自定义，必须以.repo结尾
[Server] //仓库ID
name=Server //仓库名
baseurl=file:///media/cd/Server //仓库位置 ftp：// http：//
enabled=1 //启用仓库
gpgcheck=0 //不检查软件包的签名

2. 检查目前可用的仓库
[root@yangs ~]# yum clean all //清空缓存 /var/cache/yum/x86_64/ 为yum缓存
[root@yangs ~]# yum repolist //查询可用的仓库 默认会清空缓存 修改配置文件 /etc/yum.conf 将keepcache =1

3. 安装
[root@tianyun ~]# yum list mysql-server
[root@tianyun ~]# yum install mysql-server
[root@tianyun ~]# yum -y install mysql-server
[root@tianyun ~]# yum -y install samba
[root@tianyun ~]# rpm -e samba
[root@tianyun ~]# yum -y remove mysql-server
[root@tianyun ~]# yum -y install "mysql*" httpd vsftpd samba dovecot
[root@tianyun ~]# yum -y update samba
[root@tianyun ~]# yum -y update

RPM包签名检查机制：
＝＝RPM工具使用签名检查机制：＝＝
# rpm --import /etc/pki/rpm-gpg/RPM-GPG-KEY-redhat-release //导入红帽公钥
# rpm --import /etc/pki/rpm-gpg/*
# rpm -ivh tftp-server-0.49-2.i386.rpm
Preparing... ######################################### [100%]
1:tftp-server ######################################### [100%]
额外选项：
--nomd5 //不检验软件包的签名

＝＝YUM使用签名检查机制：＝＝
方法一：
[root@tianyun yum.repos.d]# rpm --import /etc/pki/rpm-gpg/RPM-GPG-KEY-redhat-release
[root@tianyun yum.repos.d]# vim rhel6.repo
[Server]
name=Server
baseurl=file:///media/Server
enabled=1
gpgcheck=1 //是否检查软件包的签名

方法二：
[root@tianyun yum.repos.d]# vim rhel6.repo
[Server]
name=Server
baseurl=file:///media/Server
enabled=1
gpgcheck=1 //是否检查软件包的签名
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-redhat-release //指定公钥文件

额外选项：
--nogpgcheck //不检查软件包的签名
