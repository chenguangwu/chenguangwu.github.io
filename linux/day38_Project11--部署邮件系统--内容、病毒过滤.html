三、配置ClamAV
[root@tianyun ~]# yum -y install clam*

[root@tianyun ~]# vim /etc/clamd.conf
# Default: disabled
LocalSocket /var/run/clamav/clamd.sock
#TCPSocket 3310

设置相关目录权限：
[root@tianyun ~]# gpasswd -a clamav amavis
[root@tianyun ~]# chown amavis.amavis /var/spool/vscan
[root@tianyun ~]# chmod 750 /var/spool/vscan
[root@tianyun ~]# chown amavis.amavis /var/spool/vscan/tmp
[root@tianyun ~]# chmod 750 /var/spool/vscan/tmp

启动ClamAV及开机启动：
[root@tianyun ~]# service clamd start
Starting Clam AntiVirus Daemon: LibClamAV Warning: **************************************************
LibClamAV Warning: *** The virus database is older than 7 days! ***
LibClamAV Warning: *** Please update it as soon as possible. ***
LibClamAV Warning: **************************************************
[ OK ]
[root@tianyun ~]# freshclam -daemon
[root@tianyun ~]# chkconfig clamd on

四、配置amavisd.conf
[root@tianyun ~]# vim /etc/amavisd/amavisd.conf
$max_servers = 2; //设置最大可使用的进程数
$lock_file = "$MYHOME/amavisd.lock";
$pid_file = "$MYHOME/amavisd.pid";
$mydomain = 'tianyun.com';
$myhostname = 'mail.tianyun.com';
@local_domains_maps = qw(.); //对所有域检查

修改投递/拦截的方法：
$sa_tag2_level_deflt = 6.2; //达到6.2分，在邮件标题加入[SPAM]标记
$sa_kill_level_deflt = 6.9; //达到6.9分，直接删除
$final_virus_destiny = D_DISCARD; //病毒邮件处理方法
$final_banned_destiny = D_DISCARD;
$final_spam_destiny = D_PASS; //垃圾邮件处理方法
$final_bad_header_destiny = D_PASS;

配置Amavisd与ClamAV结合：
添加：
@av_scanners = (
['ClamAV-clamd',
\&ask_daemon, ["CONTSCAN {}\n", "/var/run/clamav/clamd.sock"],
qr/\bOK$/, qr/\bFOUND$/,
qr/^.*?: (?!Infected Archive)(.*) FOUND$/ ],
);

@av_scanners_backup = (
['ClamAV-clamscan', 'clamscan',
"--stdout --no-summary -r --tempdir=$TEMPBASE {}",
[0], qr/:.*\sFOUND$/, qr/^.*?: (?!Infected Archive)(.*) FOUND$/ ],
);


五、配置Postfix集成amavisd-new
1. 增加邮件别名
[root@tianyun ~]# vim /etc/postfix/aliases
virusalert: root
spam.police: root
postfix: jack@tianyun.com
[root@tianyun ~]# newaliases
[root@tianyun ~]# service amavisd restart

2. 编辑master.conf
[root@tianyun ~]# vim /etc/postfix/master.cf
smtp-amavis unix - - n - 3 smtp
-o smtp_data_done_timeout=1200
-o smtp_send_xforward_command=yes
-o disable_dns_lookups=yes
-o max_use=10

127.0.0.1:10025 inet n - n - - smtpd
-o content_filter=
-o local_recipient_maps=
-o relay_recipient_maps=
-o smtpd_restriction_classes=
-o smtpd_client_restrictions=
-o smtpd_helo_restrictions=
-o smtpd_sender_restrictions=
-o smtpd_recipient_restrictions=permit_mynetworks,reject
-o mynetworks=127.0.0.0/8
-o strict_rfc821_envelopes=yes
-o smtpd_error_sleep_time=0
-o smtpd_soft_error_limit=1001
-o smtpd_hard_error_limit=1000
-o receive_override_options=no_unknown_recipient_checks,no_header_body_checks

3. 编辑main.conf
# Content-Filter
content_filter = smtp-amavis:[127.0.0.1]:10024
receive_override_options = no_address_mappings

4. 重启
[root@tianyun ~]# service postfix restart
[root@tianyun ~]# service amavisd restart
[root@tianyun ~]# service clamd restart
