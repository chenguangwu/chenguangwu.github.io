HttpServletRequestWrapper模拟实现分布式Session

HttpSession的内容都放在一个单独的Map中，模拟远程分布式Session。

1.使用HttpServletRequestWrapper创建自定义Request
2.使用动态代理包装自定义Request返回的HttpSession对象
3.创建过滤器，使用自定义Request替换原有的Request对象。
4.在Servlet中得到的HttpSession对象，写入和读取内容都假设通过远程Session服务器。

创建自定义的Request，返回动态代理的HttpSession

  2. import java.lang.reflect.Method;
  3. import java.lang.reflect.Proxy;

  14. public class RemoteSessionRequest extends HttpServletRequestWrapper {

  16.     public RemoteSessionRequest(HttpServletRequest request) {
  17.         super(request);
  18.     }

  20.     @Override

  22.         return RemoteSessionHandler.getInstance(super.getSession());
  23.     }

  26. class RemoteSessionHandler implements InvocationHandler {
  27.     //模拟远程Session服务器，Key表示SessionId,Value表示该Session的内容
  28.     private static Map<String, Map<String, Object>> map = new ConcurrentHashMap<String, Map<String, Object>>();

  30.     private HttpSession session = null;

  32.     private RemoteSessionHandler(HttpSession httpSession) {
  33.         this.session = httpSession;
  34.     };

  36.     public static HttpSession getInstance(HttpSession httpSession) {
  37.         InvocationHandler handler = new RemoteSessionHandler(httpSession);
  38.         return (HttpSession) Proxy.newProxyInstance(httpSession.getClass().getClassLoader(), httpSession.getClass().getInterfaces(), handler);
  39.     }

  41.     @Override
  42.     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
  43.         if ("setAttribute".equals(method.getName())) {
  44.             String id = session.getId();
  45.             Map<String, Object> m = map.get(id);
  46.             if (m == null) {
  47.                 m = new HashMap<String, Object>();
  48.                 map.put(id, m);
  49.             }
  50.             m.put((String) args[0], args[1]);
  51.             System.out.println("[存入]key:" + args[0] + ",value:" + args[1]);
  52.             return null;
  53.         } else if ("getAttribute".equals(method.getName())) {
  54.             String id = session.getId();
  55.             Map<String, Object> m = map.get(id);
  56.             if (m == null) {
  57.                 return null;
  58.             }
  59.             Object result = m.get(args[0]);
  60.             System.out.println("[取出]key:" + args[0] + ",value:" + result);
  61.             return result;
  62.         }
  63.         return method.invoke(session, args);
  64.     }

  66. }
