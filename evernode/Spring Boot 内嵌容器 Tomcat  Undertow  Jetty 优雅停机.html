Spring Boot 内嵌容器 Tomcat / Undertow / Jetty 优雅停机实现

2018-04-19 Anoyi SpringForAll社区

Spring Boot 在关闭时，如果有请求没有响应完，在不同的容器会出现不同的结果，例如，在 Tomcat 和 Undertow 中会出现中断异常，那么就有可能对业务造成影响。所以，优雅停机非常有必要性，目前官方是没有提供很好的策略来实现。

Each SpringApplication registers a shutdown hook with the JVM to ensure that the ApplicationContext closes gracefully on exit. All the standard Spring lifecycle callbacks (such as the DisposableBean interface or the @PreDestroy annotation) can be used.

Spring Boot Application 在接收到停机信号后，可以通过 DisposableBean 接口 、 @PreDestroy 注解 或者 ContextClosedEvent 事件来处理优雅停机的相关逻辑。

版本信息

Spring Boot 版本： 2.0.0.RELEASE

Tomcat 优雅停机

  1. @SpringBootApplication
  2. public class Application {
  4.     public static void main(String[] args) {
  5.         SpringApplication.run(Application.class, args);
  6.     }
  8.     /**
  9.      * 用于接受 shutdown 事件
  10.      */
  11.     @Bean
  12.     public GracefulShutdown gracefulShutdown() {
  13.         return new GracefulShutdown();
  14.     }
  16.     @Bean
  17.     public ServletWebServerFactory servletContainer() {
  18.         TomcatServletWebServerFactory tomcat = new TomcatServletWebServerFactory();
  19.         tomcat.addConnectorCustomizers(gracefulShutdown());
  20.         return tomcat;
  21.     }
  23.     /**
  24.      * 优雅关闭 Spring Boot
  25.      */
  26.     private class GracefulShutdown implements TomcatConnectorCustomizer, ApplicationListener<ContextClosedEvent> {
  27.         private final Logger log = LoggerFactory.getLogger(GracefulShutdown.class);
  28.         private volatile Connector connector;
  29.         private final int waitTime = 30;
  30.         @Override
  31.         public void customize(Connector connector) {
  32.             this.connector = connector;
  33.         }
  34.         @Override
  35.         public void onApplicationEvent(ContextClosedEvent contextClosedEvent) {
  36.             this.connector.pause();
  37.             Executor executor = this.connector.getProtocolHandler().getExecutor();
  38.             if (executor instanceof ThreadPoolExecutor) {
  39.                 try {
  40.                     ThreadPoolExecutor threadPoolExecutor = (ThreadPoolExecutor) executor;
  41.                     threadPoolExecutor.shutdown();
  42.                     if (!threadPoolExecutor.awaitTermination(waitTime, TimeUnit.SECONDS)) {
  43.                         log.warn("Tomcat thread pool did not shut down gracefully within " + waitTime + " seconds. Proceeding with forceful shutdown");
  44.                     }
  45.                 } catch (InterruptedException ex) {
  46.                     Thread.currentThread().interrupt();
  47.                 }
  48.             }
  49.         }
  50.     }
  52. }

Undertow 优雅停机

  1. @SpringBootApplication
  2. public class Application {
  4.     public static void main(String[] args) {
  5.         SpringApplication.run(Application.class, args);
  6.     }
  8.     /**
  9.      * 优雅关闭 Spring Boot
  10.      */
  11.     @Component
  12.     public class GracefulShutdown implements ApplicationListener<ContextClosedEvent> {
  14.         @Autowired
  15.         private GracefulShutdownWrapper gracefulShutdownWrapper;
  17.         @Autowired
  18.         private ServletWebServerApplicationContext context;
  20.         @Override
  21.         public void onApplicationEvent(ContextClosedEvent contextClosedEvent){
  22.             gracefulShutdownWrapper.getGracefulShutdownHandler().shutdown();
  23.             try {
  24.                 UndertowServletWebServer webServer = (UndertowServletWebServer)context.getWebServer();
  25.                 Field field = webServer.getClass().getDeclaredField("undertow");
  26.                 field.setAccessible(true);
  27.                 Undertow undertow = (Undertow) field.get(webServer);
  28.                 List<Undertow.ListenerInfo> listenerInfo = undertow.getListenerInfo();
  29.                 Undertow.ListenerInfo listener = listenerInfo.get(0);
  30.                 ConnectorStatistics connectorStatistics = listener.getConnectorStatistics();
  31.                 while (connectorStatistics.getActiveConnections() > 0){}
  32.             }catch (Exception e){
  33.                 // Application Shutdown
  34.             }
  35.         }
  36.     }
  37. }
  1. @Component
  2. public class GracefulShutdownWrapper implements HandlerWrapper{
  4.     private GracefulShutdownHandler gracefulShutdownHandler;
  6.     @Override
  7.     public HttpHandler wrap(HttpHandler handler) {
  8.         if(gracefulShutdownHandler == null) {
  9.             this.gracefulShutdownHandler = new GracefulShutdownHandler(handler);
  10.         }
  11.         return gracefulShutdownHandler;
  12.     }
  14.     public GracefulShutdownHandler getGracefulShutdownHandler() {
  15.         return gracefulShutdownHandler;
  16.     }
  18. }
  1. @Component
  2. @AllArgsConstructor
  3. public class UndertowExtraConfiguration {
  5.     private final GracefulShutdownWrapper gracefulShutdownWrapper;
  7.     @Bean
  8.     public UndertowServletWebServerFactory servletWebServerFactory() {
  9.         UndertowServletWebServerFactory factory = new UndertowServletWebServerFactory();
  10.         factory.addDeploymentInfoCustomizers(deploymentInfo -> deploymentInfo.addOuterHandlerChainWrapper(gracefulShutdownWrapper));
  11.         factory.addBuilderCustomizers(builder -> builder.setServerOption(UndertowOptions.ENABLE_STATISTICS, true));
  12.         return factory;
  13.     }
  15. }
