  1. package com.cn.fanjg.rtti;

  3. import java.lang.reflect.Field;
  4. import java.lang.reflect.InvocationTargetException;
  5. import java.lang.reflect.Method;
  6. import java.util.regex.Matcher;
  7. import java.util.regex.Pattern;

  9. /**
  10.  * @function：
  11.  * @date : 2013-4-10 下午03:28:53
  12.  */
  13. @SuppressWarnings("rawtypes")
  14. public class AssignValueForAttributeUtil {
  15.     /** 正则表达式 用于匹配属性的第一个字母 {@value [a-zA-Z]} **/
  16.     private static final String REGEX = "[a-zA-Z]";
  17.     public static void setAttrributeValue(Object obj,String attribute,Object value)
  18.     {
  19.         String method_name = convertToMethodName(attribute,obj.getClass(),true);
  20.         Method[] methods = obj.getClass().getMethods();
  21.         for (Method method : methods) {
  22.             /**
  23.              *     因为这里只是调用bean中属性的set方法，属性名称不能重复
  24.              * 所以set方法也不会重复，所以就直接用方法名称去锁定一个方法
  25.              * （注：在java中，锁定一个方法的条件是方法名及参数）
  26.              * **/
  27.             if(method.getName().equals(method_name))
  28.             {
  29.                 Class[] parameterC = method.getParameterTypes();
  30.                 try {
  31.                     /**如果是基本数据类型时（如int、float、double、byte、char、boolean）
  32.                      * 需要先将Object转换成相应的封装类之后再转换成对应的基本数据类型
  33.                      * 否则会报 ClassCastException**/
  34.                     if(parameterC[0] == int.class)
  35.                     {
  36.                         method.invoke(obj,((Integer)value).intValue());
  37.                         break;
  38.                     }else if(parameterC[0] == float.class){
  39.                         method.invoke(obj, ((Float)value).floatValue());
  40.                         break;
  41.                     }else if(parameterC[0] == double.class)
  42.                     {
  43.                         method.invoke(obj, ((Double)value).doubleValue());
  44.                         break;
  45.                     }else if(parameterC[0] == byte.class)
  46.                     {
  47.                         method.invoke(obj, ((Byte)value).byteValue());
  48.                         break;
  49.                     }else if(parameterC[0] == char.class)
  50.                     {
  51.                         method.invoke(obj, ((Character)value).charValue());
  52.                         break;
  53.                     }else if(parameterC[0] == boolean.class)
  54.                     {
  55.                         method.invoke(obj, ((Boolean)value).booleanValue());
  56.                         break;
  57.                     }else
  58.                     {
  59.                         method.invoke(obj,parameterC[0].cast(value));
  60.                         break;
  61.                     }
  62.                 } catch (IllegalArgumentException e) {
  63.                     e.printStackTrace();
  64.                 } catch (IllegalAccessException e) {
  65.                     e.printStackTrace();
  66.                 } catch (InvocationTargetException e) {
  67.                     e.printStackTrace();
  68.                 } catch (SecurityException e) {
  69.                     e.printStackTrace();
  70.                 }
  71.             }
  72.         }
  73.     }
  74.     private static String convertToMethodName(String attribute,Class objClass,boolean isSet)
  75.     {
  76.         /** 通过正则表达式来匹配第一个字符 **/
  77.         Pattern p = Pattern.compile(REGEX);
  78.         Matcher m = p.matcher(attribute);
  79.         StringBuilder sb = new StringBuilder();
  80.         /** 如果是set方法名称 **/
  81.         if(isSet)
  82.         {
  83.             sb.append("set");
  84.         }else{
  85.         /** get方法名称 **/
  86.             try {
  87.                 Field attributeField = objClass.getDeclaredField(attribute);
  88.                 /** 如果类型为boolean **/
  89.                 if(attributeField.getType() == boolean.class||attributeField.getType() == Boolean.class)
  90.                 {
  91.                     sb.append("is");
  92.                 }else
  93.                 {
  94.                     sb.append("get");
  95.                 }
  96.             } catch (SecurityException e) {
  97.                 e.printStackTrace();
  98.             } catch (NoSuchFieldException e) {
  99.                 e.printStackTrace();
  100.             }
  101.         }
  102.         /** 针对以下划线开头的属性 **/
  103.         if(attribute.charAt(0)!='_' && m.find())
  104.         {
  105.             sb.append(m.replaceFirst(m.group().toUpperCase()));
  106.         }else{
  107.             sb.append(attribute);
  108.         }
  109.         return sb.toString();
  110.     }
  111.     public static Object getAttrributeValue(Object obj,String attribute)
  112.     {
  113.         String methodName = convertToMethodName(attribute, obj.getClass(), false);
  114.         Object value = null;
  115.         try {
  116.             /** 由于get方法没有参数且唯一，所以直接通过方法名称锁定方法 **/
  117.             Method methods = obj.getClass().getDeclaredMethod(methodName);
  118.             if(methods != null)
  119.             {
  120.                 value = methods.invoke(obj);
  121.             }
  122.         } catch (SecurityException e) {
  123.             e.printStackTrace();
  124.         } catch (NoSuchMethodException e) {
  125.             e.printStackTrace();
  126.         } catch (IllegalArgumentException e) {
  127.             e.printStackTrace();
  128.         } catch (IllegalAccessException e) {
  129.             e.printStackTrace();
  130.         } catch (InvocationTargetException e) {
  131.             e.printStackTrace();
  132.         }
  133.         return value;
  134.     }
