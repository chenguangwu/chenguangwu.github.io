代码案例：

复制代码 代码如下:

<script type="text/javascript">
//取整
function getResult(num){
return parseInt(num);
}

//四舍五入到num后面的n位
function getResult(num,n){
return Math.round(num*Math.pow(10,n))/Math.pow(10,n);
}

//截取n位
function getresult(num,n){
return num.toString().replace(new RegExp("^(\\-?\\d*\\.?\\d{0,"+n+"})(\\d*)$"),"$1")+0;
}
</script>
