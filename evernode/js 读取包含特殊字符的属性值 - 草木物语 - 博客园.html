js 读取包含特殊字符的属性值

 

 

在JS中对象的属性可以通过两种方式访问：object.property和object["property"]。

包含特殊字符的属性只能以此方式访问: object["property"]。

// 在JS中对象的属性可以通过两种方式访问：object.property和object["property"]。
var obj = new Object();
obj.name = "CodePlayer";
obj.age = 18;
obj.sayHi = function() {
    alert("Hello World!");
};

// 包含特殊字符的属性只能以此方式访问
obj["foo-bar"] = "包含特殊字符";

console.log(obj.age); // 18
console.log(obj['age']); // 18
console.log(obj["foo-bar"]); // 包含特殊字符
